		weak(\f,\mu,\d,\sig) = sqrt((1+2/3*\d/0.1*(\mu-1)^2/\mu)^2+(2/3*0.1*\d/\mu/(3*10^8/sqrt(2*pi*\sig*\mu*2*pi*\f))^2)^2);
	}
]
	\begin{loglogaxis}[
	table/col sep = semicolon,
